//crie uma calculadora que irá capturar qual operação aritmética o usuario deseja fazer //utilize switch-case para resolver o problema //entrada de dados let num1 = parseInt(prompt("Informe o primeiro número: ")) let num2 = parseInt(prompt("Informe o segundo número: ")) //menu opções const escolhaUser = Number(prompt(` Escolha uma operação: [1] - soma [2] - subtração [3] - divisão [4] - multiplicação `)) //processamento //realizar o calculo com base na opção que o usuario digitou switch (escolhaUser) { case 1: const calculoSoma = num1 + num2 alert(`O resultado de ${num1} + ${num2} = ${calculoSoma}`) break; case 2: const calculoSub = num1 - num2 alert(`O resultado de ${num1} - ${num2} = ${calculoSub}`) break; case 3: const calculoDiv = num1 / num2 num1 == 0 || num2 == 0 ? alert("Não é possível dividir por 0") : alert(`O resultado de ${num1} / ${num2} = ${calculoDiv}`) break; case 4: const calculoMult = num1 * num2 alert(`O resultado de ${num1} * ${num2} = ${calculoMult}`) break; default: alert("opção inválida!") break; } alert("obrigado por usar a minha calculadora!")
